#For each of the two samples compute the posterior 
 
results <- data.frame(matrix(nrow = 0, ncol = 5))
colnames(results) <- c("Mean", "Var", "Median", "CI_lower", "CI_upper")
corps <- c("n1", "n2")

par(mfrow = c(1, 2))  # Plotting space
colors <- c("firebrick", "blue")
for(i in 1:ncol(df)){
    #Posterior will be a Gamma function -> find parameters
    alpha <- 1 + sum(df[,i] * seq(from = 0, to = 5, by = 1))
    lambda <- sum(df[,i])

    #Evaluate mean, median, variance
    mean <- integrate(function(x) dgamma(x, shape = alpha, scale = 1/lambda)*x,0,Inf)$value
    median <- qgamma(0.5, shape = alpha, scale = 1/lambda)
    variance <- integrate(function(x) dgamma(x, shape = alpha, scale = 1/lambda)*(x-mean)^2, 0, Inf)$value
                          
    #Credibility interval
    CI <- c(qgamma(0.025, shape = alpha, scale = 1/lambda), qgamma(0.975, shape = alpha, scale = 1/lambda))
    
    #Store values
    results[nrow(results) + 1, ] <- c(mean, variance, median, CI[1], CI[2])                   
                          
    #Plots 
    xs <- seq(from = 0, to = 1, length.out = 1001)        # x points
    yplot <- dgamma(xs, shape = alpha, scale = 1/lambda)  # y points
                          
    plot(xs, yplot, type = "l", main = paste("Posterior after uniform prior, ", corps[i]),
        col = colors[i], lwd = 2, xlab = "\u03bb", ylab = "p(\u03bb)", cex.lab = 1.3, cex.main = 1) #curve
                          
    segments(x0 = mean, y0 = 0, x1 = mean, y1 = dgamma(mean, shape = alpha, scale = 1/lambda),
             col="black", lty = 2, lwd = 2) #mean
                          
    color = col2rgb(colors[i], alpha = FALSE)[,1]
    polygon(c(CI[2], CI[2], CI[1], CI[1], xs[xs > CI[1] & xs < CI[2]]),
            c(dgamma(CI[2], shape = alpha, scale = 1/lambda),0, 0, dgamma(CI[1], shape = alpha, scale = 1/lambda), yplot[xs > CI[1] & xs < CI[2]]), 
            col =rgb(color[1]/255, color[2]/255, color[3]/255, alpha = 0.2))    #shade CI under curve

    #Legend
    legend("topleft", legend = "95% CI",
           fill = rgb(color[1]/255, color[2]/255, color[3]/255, alpha = 0.2) )

}

d) Test the hypotesis H0 : $p = 0.1$ versus H1 : $p\neq0.1$ at 5% level of significance with both the frequentist and bayesian approach¶

Frequentist approach¶

$H_0: p = 0.1$ $H_1: p!= 0.1$ We measure $y$ contaminated samples. If $H_0$ holds true, then the distribution of $y$ will be a binomial with $p=0.1$.

In [20]:
#I write it as a function, because I will also need it later in point h.
frequentist_approach <- function(y_func,n_func, color){
    xs <- seq(from = 0, to = n_func, by = 1)
    values <- dbinom(xs, size = n_func, prob = 0.1) #binomial distribution
    df <- data.frame(xs, values)

    #Two sided --> find approximately 0.025 on one side and 0.975 on the other
    cumulatives <- rep(0, length(values)) 
    cumulatives[1] = values[1]
    for(i in 2:length(values)){
        cumulatives[i] <- cumulatives[i-1] + values[i]
    }
    cumulatives <- unlist(cumulatives)

    ##Find first entry in cumulatives >= 0.025, and check the previous value. Choose the value closest to 0.025
    indeces_min <- c(min(which(cumulatives >= 0.025))-1, min(which(cumulatives >= 0.025)))
    indeces_max <- c(max(which(cumulatives <= 0.975)), max(which(cumulatives <= 0.975))+1)
    
    if(abs(cumulatives[indeces_min[1]] - 0.025) < abs(cumulatives[indeces_min[2]]-0.025)){
        indeces_min <- indeces_min[1]
    }
    else{
        indeces_min <- indeces_min[2]
    }
    
    if(abs(cumulatives[indeces_max[1]] - 0.975) < abs(cumulatives[indeces_max[2]]-0.975)){
        indeces_max <- indeces_max[1]
    }
    else{
        indeces_max <- indeces_max[2]
    }
    
    x_min <- xs[indeces_min]
    x_max <- xs[indeces_max]

    alpha_used = cumulatives[indeces_min] + (1 - cumulatives[indeces_max])
    
    print(paste("First point in acceptance region: ", x_min))
    print(paste("Last point in acceptance region: ", x_max))
    
    print(paste("Alpha used: ", alpha_used))
    
    print(paste("Lower tail integral: ", cumulatives[indeces_min]))
    print(paste("Upper tail integral: ", cumulatives[indeces_max]))
    
    #Plot and show where our observation lands. Change colour to highlight acceptance region
    x_barplot = seq(from = 0, to = 30, by = 1)
    
    colours = rep("white", length(x_barplot))                # only plot from 0 to x_max + 5 to "zoom in"
    colours[x_barplot >= x_min & x_barplot <= x_max] = color # acceptance region
    
    barplot(dbinom(x_barplot, n_func, 0.1), col = colours, 
            names.arg = seq(from = 0, to = x_barplot[length(x_barplot)], by = 1), 
            main = paste("Binomial distribution with p = 0.1, n = ", n_func, ", observation = ", y_func),
           xlab="y", ylab="P(y|p = 0.1)", cex.lab = 1.2, las=2) #binomial, restricted to [0,35] for plotting
    
    #Add legend
    legend("topright", legend = c("Acceptance region", "Rejection region"), fill = c(color, "white") )
    
}
